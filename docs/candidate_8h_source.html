76  // TODO(pthatcher): Remove once Chromium's jingle/glue/utils.cc
77  // doesn't use it.
78  // Maps old preference (which was 0.0-1.0) to match priority (which
79  // is 0-2^32-1) to to match RFC 5245, section 4.1.2.1. Also see
80  // https://docs.google.com/a/google.com/document/d/
81  // 1iNQDiwDKMh0NQOrCqbj3DKKRT0Dn5_5UJYhmZO-t7Uc/edit
82  float preference() const {
83  // The preference value is clamped to two decimal precision.
84  return static_cast<float>(((priority_ >> 24) * 100 / 127) / 100.0);
85  }
86 
87  // TODO(pthatcher): Remove once Chromium's jingle/glue/utils.cc
88  // doesn't use it.
89  void set_preference(float preference) {
90  // Limiting priority to UINT_MAX when value exceeds uint32_t max.
91  // This can happen for e.g. when preference = 3.
92  uint64_t prio_val = static_cast<uint64_t>(preference * 127) << 24;
93  priority_ = static_cast<uint32_t>(
94  std::min(prio_val, static_cast<uint64_t>(UINT_MAX)));
95  }
