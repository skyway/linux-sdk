30 // Interpolated gain curve using under-approximation to avoid saturation.
31 //
32 // The goal of this class is allowing fast look ups to get an accurate
33 // estimates of the gain to apply given an estimated input level.
34 class InterpolatedGainCurve {
35  public:
36  enum class GainCurveRegion {
37  kIdentity = 0,
38  kKnee = 1,
39  kLimiter = 2,
40  kSaturation = 3
41  };
42 
43  struct Stats {
44  // Region in which the output level equals the input one.
45  size_t look_ups_identity_region = 0;
46  // Smoothing between the identity and the limiter regions.
47  size_t look_ups_knee_region = 0;
48  // Limiter region in which the output and input levels are linearly related.
49  size_t look_ups_limiter_region = 0;
50  // Region in which saturation may occur since the input level is beyond the
51  // maximum expected by the limiter.
52  size_t look_ups_saturation_region = 0;
53  // True if stats have been populated.
54  bool available = false;
55 
56  // The current region, and for how many frames the level has been
57  // in that region.
58  GainCurveRegion region = GainCurveRegion::kIdentity;
59  int64_t region_duration_frames = 0;
60  };
