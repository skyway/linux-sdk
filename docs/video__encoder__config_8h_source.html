29 // The `VideoStream` struct describes a simulcast layer, or "stream".
30 struct VideoStream {
31  VideoStream();
32  ~VideoStream();
33  VideoStream(const VideoStream& other);
34  std::string ToString() const;
35 
36  // Width/Height in pixels.
37  // This is the actual width and height used to configure encoder,
38  // which might be less than `requested_resolution` due to adaptation
39  // or due to the source providing smaller frames than requested.
40  size_t width;
41  size_t height;
42 
43  // Frame rate in fps.
44  int max_framerate;
45 
46  // Bitrate, in bps, for the stream.
47  int min_bitrate_bps;
48  int target_bitrate_bps;
49  int max_bitrate_bps;
50 
51  // Scaling factor applied to the stream size.
52  // `width` and `height` values are already scaled down.
53  double scale_resolution_down_by;
54 
55  // Maximum Quantization Parameter to use when encoding the stream.
56  int max_qp;
57 
58  // Determines the number of temporal layers that the stream should be
59  // encoded with. This value should be greater than zero.
60  // TODO(brandtr): This class is used both for configuring the encoder
61  // (meaning that this field _must_ be set), and for signaling the app-level
62  // encoder settings (meaning that the field _may_ be set). We should separate
63  // this and remove this optional instead.
64  absl::optional<size_t> num_temporal_layers;
65 
66  // The priority of this stream, to be used when allocating resources
67  // between multiple streams.
68  absl::optional<double> bitrate_priority;
69 
70  absl::optional<ScalabilityMode> scalability_mode;
71 
72  // If this stream is enabled by the user, or not.
73  bool active;
74 
75  // An optional user supplied max_frame_resolution
76  // than can be set independently of (adapted) VideoSource.
77  // This value is set from RtpEncodingParameters::requested_resolution
78  // (i.e. used for signaling app-level settings).
79  //
80  // The actual encode resolution is in `width` and `height`,
81  // which can be lower than requested_resolution,
82  // e.g. if source only provides lower resolution or
83  // if resource adaptation is active.
84  absl::optional<Resolution> requested_resolution;
85 };

180  // Padding will be used up to this bitrate regardless of the bitrate produced
181  // by the encoder. Padding above what's actually produced by the encoder helps
182  // maintaining a higher bitrate estimate. Padding will however not be sent
183  // unless the estimated bandwidth indicates that the link can handle it.
184  int min_transmit_bitrate_bps;
185  int max_bitrate_bps;
186  // The bitrate priority used for all VideoStreams.
187  double bitrate_priority;
188 
189  // The simulcast layer's configurations set by the application for this video
190  // sender. These are modified by the video_stream_factory before being passed
191  // down to lower layers for the video encoding.
192  // `simulcast_layers` is also used for configuring non-simulcast (when there
193  // is a single VideoStream).
194  // We have the same number of `simulcast_layers` as we have negotiated
195  // encodings, for example 3 are used in both simulcast and legacy kSVC.
196  std::vector<VideoStream> simulcast_layers;
197 
198  // Max number of encoded VideoStreams to produce.
199  // This is the same as the number of encodings negotiated (i.e. SSRCs),
200  // whether or not those encodings are `active`, except for when legacy kSVC
201  // is used. In this case we have three SSRCs but `number_of_streams` is
202  // changed to 1 to tell lower layers to limit the number of streams.
203  size_t number_of_streams;
204 
205  // Legacy Google conference mode flag for simulcast screenshare
206  bool legacy_conference_mode;
207 
208  // Indicates whether quality scaling can be used or not.
209  bool is_quality_scaling_allowed;
210 
211  // Maximum Quantization Parameter.
212  // This value is fed into EncoderStreamFactory that
213  // apply it to all simulcast layers/spatial layers.
214  int max_qp;
215 
216  private:
217  // Access to the copy constructor is private to force use of the Copy()
218  // method for those exceptional cases where we do use it.
219  VideoEncoderConfig(const VideoEncoderConfig&);
220 };
