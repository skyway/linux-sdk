162  // The actual number of simulcast streams. This is <= 1 in singlecast (it can
163  // be 0 in old code paths), but it is also 1 in the {active,inactive,inactive}
164  // "single RTP simulcast" use case and the legacy kSVC use case. In all other
165  // cases this is the same as the number of encodings (which may include
166  // inactive encodings). In other words:
167  // - `numberOfSimulcastStreams <= 1` in singlecast and singlecast-like setups
168  // including legacy kSVC (encodings interpreted as spatial layers) or
169  // standard kSVC (1 active encoding).
170  // - `numberOfSimulcastStreams > 1` in simulcast of 2+ active encodings.
171  unsigned char numberOfSimulcastStreams;
172  SimulcastStream simulcastStream[kMaxSimulcastStreams];
173  SpatialLayer spatialLayers[kMaxSpatialLayers];
174 
175  VideoCodecMode mode;
176  bool expect_encode_from_texture;
177 
178  // Timing frames configuration. There is delay of delay_ms between two
179  // consequent timing frames, excluding outliers. Frame is always made a
180  // timing frame if it's at least outlier_ratio in percent of "ideal" average
181  // frame given bitrate and framerate, i.e. if it's bigger than
182  // |outlier_ratio / 100.0 * bitrate_bps / fps| in bits. This way, timing
183  // frames will not be sent too often usually. Yet large frames will always
184  // have timing information for debug purposes because they are more likely to
185  // cause extra delays.
186  struct TimingFrameTriggerThresholds {
187  int64_t delay_ms;
188  uint16_t outlier_ratio_percent;
189  } timing_frame_thresholds;
190 
191  // Legacy Google conference mode flag for simulcast screenshare
192  bool legacy_conference_mode;

218  bool frame_drop_enabled_ = false;
219 };
