36 enum TemporalStructureMode {
37  kTemporalStructureMode1, // 1 temporal layer structure - i.e., IPPP...

39  kTemporalStructureMode3, // 3 temporal layers 0212...

42 struct GofInfoVP9 {
43  void SetGofInfoVP9(TemporalStructureMode tm) {
44  switch (tm) {
45  case kTemporalStructureMode1:
46  num_frames_in_gof = 1;
47  temporal_idx[0] = 0;
48  temporal_up_switch[0] = true;
49  num_ref_pics[0] = 1;
50  pid_diff[0][0] = 1;
51  break;
52  case kTemporalStructureMode2:
53  num_frames_in_gof = 2;
54  temporal_idx[0] = 0;
55  temporal_up_switch[0] = true;
56  num_ref_pics[0] = 1;
57  pid_diff[0][0] = 2;
58 
59  temporal_idx[1] = 1;
60  temporal_up_switch[1] = true;
61  num_ref_pics[1] = 1;
62  pid_diff[1][0] = 1;
63  break;
64  case kTemporalStructureMode3:
65  num_frames_in_gof = 4;
66  temporal_idx[0] = 0;
67  temporal_up_switch[0] = true;
68  num_ref_pics[0] = 1;
69  pid_diff[0][0] = 4;
70 
71  temporal_idx[1] = 2;
72  temporal_up_switch[1] = true;
73  num_ref_pics[1] = 1;
74  pid_diff[1][0] = 1;
75 
76  temporal_idx[2] = 1;
77  temporal_up_switch[2] = true;
78  num_ref_pics[2] = 1;
79  pid_diff[2][0] = 2;
80 
81  temporal_idx[3] = 2;
82  temporal_up_switch[3] = true;
83  num_ref_pics[3] = 1;
84  pid_diff[3][0] = 1;
85  break;
86  default:
87  RTC_DCHECK_NOTREACHED();
88  }
89  }

91  void CopyGofInfoVP9(const GofInfoVP9& src) {
92  num_frames_in_gof = src.num_frames_in_gof;
93  for (size_t i = 0; i < num_frames_in_gof; ++i) {
94  temporal_idx[i] = src.temporal_idx[i];
95  temporal_up_switch[i] = src.temporal_up_switch[i];
96  num_ref_pics[i] = src.num_ref_pics[i];
97  for (uint8_t r = 0; r < num_ref_pics[i]; ++r) {
98  pid_diff[i][r] = src.pid_diff[i][r];
99  }
100  }
101  }
102 
103  friend bool operator==(const GofInfoVP9& lhs, const GofInfoVP9& rhs) {
104  if (lhs.num_frames_in_gof != rhs.num_frames_in_gof ||
105  lhs.pid_start != rhs.pid_start)
106  return false;
107  for (size_t i = 0; i < lhs.num_frames_in_gof; ++i) {
108  if (lhs.temporal_idx[i] != rhs.temporal_idx[i] ||
109  lhs.temporal_up_switch[i] != rhs.temporal_up_switch[i] ||
110  lhs.num_ref_pics[i] != rhs.num_ref_pics[i]) {
111  return false;
112  }
113  for (uint8_t r = 0; r < lhs.num_ref_pics[i]; ++r) {
114  if (lhs.pid_diff[i][r] != rhs.pid_diff[i][r])
115  return false;
116  }
117  }
118  return true;
119  }
120 
121  friend bool operator!=(const GofInfoVP9& lhs, const GofInfoVP9& rhs) {
122  return !(lhs == rhs);
123  }

133 struct RTPVideoHeaderVP9 {
134  void InitRTPVideoHeaderVP9() {
135  inter_pic_predicted = false;
136  flexible_mode = false;
137  beginning_of_frame = false;
138  end_of_frame = false;
139  ss_data_available = false;
140  non_ref_for_inter_layer_pred = false;
141  picture_id = kNoPictureId;
142  max_picture_id = kMaxTwoBytePictureId;
143  tl0_pic_idx = kNoTl0PicIdx;
144  temporal_idx = kNoTemporalIdx;
145  spatial_idx = kNoSpatialIdx;
146  temporal_up_switch = false;
147  inter_layer_predicted = false;
148  gof_idx = kNoGofIdx;

151  first_active_layer = 0;

155  friend bool operator==(const RTPVideoHeaderVP9& lhs,
156  const RTPVideoHeaderVP9& rhs) {
157  if (lhs.inter_pic_predicted != rhs.inter_pic_predicted ||
158  lhs.flexible_mode != rhs.flexible_mode ||
159  lhs.beginning_of_frame != rhs.beginning_of_frame ||
160  lhs.end_of_frame != rhs.end_of_frame ||
161  lhs.ss_data_available != rhs.ss_data_available ||
162  lhs.non_ref_for_inter_layer_pred != rhs.non_ref_for_inter_layer_pred ||
163  lhs.picture_id != rhs.picture_id ||
164  lhs.max_picture_id != rhs.max_picture_id ||
165  lhs.tl0_pic_idx != rhs.tl0_pic_idx ||
166  lhs.temporal_idx != rhs.temporal_idx ||
167  lhs.spatial_idx != rhs.spatial_idx || lhs.gof_idx != rhs.gof_idx ||
168  lhs.temporal_up_switch != rhs.temporal_up_switch ||
169  lhs.inter_layer_predicted != rhs.inter_layer_predicted ||
170  lhs.num_ref_pics != rhs.num_ref_pics ||
171  lhs.end_of_picture != rhs.end_of_picture) {
172  return false;
173  }
174  for (uint8_t i = 0; i < lhs.num_ref_pics; ++i) {
175  if (lhs.pid_diff[i] != rhs.pid_diff[i] ||
176  lhs.ref_picture_id[i] != rhs.ref_picture_id[i]) {
177  return false;
178  }
179  }
180  if (lhs.ss_data_available) {
181  if (lhs.spatial_layer_resolution_present !=
182  rhs.spatial_layer_resolution_present ||
183  lhs.num_spatial_layers != rhs.num_spatial_layers ||
184  lhs.first_active_layer != rhs.first_active_layer ||
185  lhs.gof != rhs.gof) {
186  return false;
187  }
188  if (lhs.spatial_layer_resolution_present) {
189  for (size_t i = 0; i < lhs.num_spatial_layers; i++) {
190  if (lhs.width[i] != rhs.width[i] || lhs.height[i] != rhs.height[i]) {
191  return false;
192  }
193  }
194  }
195  }
196  return true;
197  }

204  bool inter_pic_predicted; // This layer frame is dependent on previously
205  // coded frame(s).
206  bool flexible_mode; // This frame is in flexible mode.
207  bool beginning_of_frame; // True if this packet is the first in a VP9 layer
208  // frame.
209  bool end_of_frame; // True if this packet is the last in a VP9 layer frame.
210  bool ss_data_available; // True if SS data is available in this payload
211  // descriptor.
212  bool non_ref_for_inter_layer_pred; // True for frame which is not used as
213  // reference for inter-layer prediction.
214  int16_t picture_id; // PictureID index, 15 bits;
215  // kNoPictureId if PictureID does not exist.
216  int16_t max_picture_id; // Maximum picture ID index; either 0x7F or 0x7FFF;
217  int16_t tl0_pic_idx; // TL0PIC_IDX, 8 bits;
218  // kNoTl0PicIdx means no value provided.
219  uint8_t temporal_idx; // Temporal layer index, or kNoTemporalIdx.
220  uint8_t spatial_idx; // Spatial layer index, or kNoSpatialIdx.
221  bool temporal_up_switch; // True if upswitch to higher frame rate is possible
222  // meaning subsequent higher temporal layer pictures
223  // will not depend on any picture before the current
224  // picture (in coding order) with temporal layer ID
225  // greater than `temporal_idx` of this frame.
226  bool inter_layer_predicted; // Frame is dependent on directly lower spatial
227  // layer frame.
228 
229  uint8_t gof_idx; // Index to predefined temporal frame info in SS data.
230 
231  uint8_t num_ref_pics; // Number of reference pictures used by this layer
232  // frame.
233  uint8_t pid_diff[kMaxVp9RefPics]; // P_DIFF signaled to derive the PictureID
234  // of the reference pictures.
235  int16_t ref_picture_id[kMaxVp9RefPics]; // PictureID of reference pictures.
236 
237  // SS data.
238  size_t num_spatial_layers; // Always populated.
239  size_t first_active_layer; // Not sent on wire, used to adjust ss data.
240  bool spatial_layer_resolution_present;
241  uint16_t width[kMaxVp9NumberOfSpatialLayers];
242  uint16_t height[kMaxVp9NumberOfSpatialLayers];
243  GofInfoVP9 gof;
244 
245  bool end_of_picture; // This frame is the last frame in picture.
246 };
