397  // DelayedMessage goes into a priority queue, sorted by trigger time. Messages
398  // with the same trigger time are processed in num_ (FIFO) order.
399  struct DelayedMessage {
400  bool operator<(const DelayedMessage& dmsg) const {
401  return (dmsg.run_time_ms < run_time_ms) ||
402  ((dmsg.run_time_ms == run_time_ms) &&
403  (dmsg.message_number < message_number));
404  }

407  int64_t run_time_ms;
408  // Monotonicaly incrementing number used for ordering of messages
409  // targeted to execute at the same time.

411  // std::priority_queue doesn't allow to extract elements, but functor
412  // is move-only and thus need to be changed when pulled out of the
413  // priority queue. That is ok because `functor` doesn't affect operator<
414  mutable absl::AnyInvocable<void() &&> functor;
