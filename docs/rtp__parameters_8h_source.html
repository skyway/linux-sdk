32 // These structures are intended to mirror those defined by:
33 // http://draft.ortc.org/#rtcrtpdictionaries*
34 // Contains everything specified as of 2017 Jan 24.
35 //
36 // They are used when retrieving or modifying the parameters of an
37 // RtpSender/RtpReceiver, or retrieving capabilities.
38 //
39 // Note on conventions: Where ORTC may use "octet", "short" and "unsigned"
40 // types, we typically use "int", in keeping with our style guidelines. The
41 // parameter's actual valid range will be enforced when the parameters are set,
42 // rather than when the parameters struct is built. An exception is made for
43 // SSRCs, since they use the full unsigned 32-bit range, and aren't expected to
44 // be used for any numeric comparisons/operations.
45 //
46 // Additionally, where ORTC uses strings, we may use enums for things that have
47 // a fixed number of supported values. However, for things that can be extended
48 // (such as codecs, by providing an external encoder factory), a string
49 // identifier is used.
50

79 // Based on the spec in
80 // https://w3c.github.io/webrtc-pc/#idl-def-rtcdegradationpreference.
81 // These options are enforced on a best-effort basis. For instance, all of
82 // these options may suffer some frame drops in order to avoid queuing.
83 // TODO(sprang): Look into possibility of more strictly enforcing the
84 // maintain-framerate option.
85 // TODO(deadbeef): Default to "balanced", as the spec indicates?
86 enum class DegradationPreference {
87  // Don't take any actions based on over-utilization signals. Not part of the
88  // web API.
89  DISABLED,
90  // On over-use, request lower resolution, possibly causing down-scaling.
91  MAINTAIN_FRAMERATE,
92  // On over-use, request lower frame rate, possibly causing frame drops.
93  MAINTAIN_RESOLUTION,
94  // Try to strike a "pleasing" balance between frame rate or resolution.
95  BALANCED,
96 };

125 struct RTC_EXPORT RtpCodec {
126  RtpCodec();
127  RtpCodec(const RtpCodec&);
128  virtual ~RtpCodec();
129 
130  // Build MIME "type/subtype" string from `name` and `kind`.
131  std::string mime_type() const { return MediaTypeToString(kind) + "/" + name; }
132 
133  // Used to identify the codec. Equivalent to MIME subtype.
134  std::string name;
135 
136  // The media type of this codec. Equivalent to MIME top-level type.
137  cricket::MediaType kind = cricket::MEDIA_TYPE_AUDIO;
138 
139  // If unset, the implementation default is used.
140  absl::optional<int> clock_rate;
141 
142  // The number of audio channels used. Unset for video codecs. If unset for
143  // audio, the implementation default is used.
144  // TODO(deadbeef): The "implementation default" part isn't fully implemented.
145  // Only defaults to 1, even though some codecs (such as opus) should really
146  // default to 2.
147  absl::optional<int> num_channels;
148 
149  // Feedback mechanisms to be used for this codec.
150  // TODO(deadbeef): Not implemented with PeerConnection senders/receivers.
151  std::vector<RtcpFeedback> rtcp_feedback;
152 
153  // Codec-specific parameters that must be signaled to the remote party.
154  //
155  // Corresponds to "a=fmtp" parameters in SDP.
156  //
157  // Contrary to ORTC, these parameters are named using all lowercase strings.
158  // This helps make the mapping to SDP simpler, if an application is using SDP.
159  // Boolean values are represented by the string "1".
160  std::map<std::string, std::string> parameters;
161 
162  bool operator==(const RtpCodec& o) const {
163  return name == o.name && kind == o.kind && clock_rate == o.clock_rate &&
164  num_channels == o.num_channels && rtcp_feedback == o.rtcp_feedback &&
165  parameters == o.parameters;
166  }
167  bool operator!=(const RtpCodec& o) const { return !(*this == o); }

195 //
196 // Just called "RtpHeaderExtension" in ORTC, but the "Capability" suffix was

211  // TODO(deadbeef): Not implemented.
212  bool preferred_encrypt = false;

215  // RtpTransceiverInterface::SetHeaderExtensionsToNegotiate() and
216  // SetHeaderExtensionsToNegotiate().
217  RtpTransceiverDirection direction = RtpTransceiverDirection::kSendRecv;
218 
219  // Constructors for convenience.
220  RtpHeaderExtensionCapability();
221  explicit RtpHeaderExtensionCapability(absl::string_view uri);
222  RtpHeaderExtensionCapability(absl::string_view uri, int preferred_id);
223  RtpHeaderExtensionCapability(absl::string_view uri,
224  int preferred_id,
225  RtpTransceiverDirection direction);
226  ~RtpHeaderExtensionCapability();
227 
228  bool operator==(const RtpHeaderExtensionCapability& o) const {
229  return uri == o.uri && preferred_id == o.preferred_id &&
230  preferred_encrypt == o.preferred_encrypt && direction == o.direction;
231  }
232  bool operator!=(const RtpHeaderExtensionCapability& o) const {
233  return !(*this == o);
234  }

239  enum Filter {
240  // Encrypted extensions will be ignored and only non-encrypted extensions

248  kRequireEncryptedExtension,

424 struct RTC_EXPORT RtpEncodingParameters {
425  RtpEncodingParameters();
426  RtpEncodingParameters(const RtpEncodingParameters&);
427  ~RtpEncodingParameters();
428 
429  // If unset, a value is chosen by the implementation.
430  //
431  // Note that the chosen value is NOT returned by GetParameters, because it
432  // may change due to an SSRC conflict, in which case the conflict is handled
433  // internally without any event. Another way of looking at this is that an
434  // unset SSRC acts as a "wildcard" SSRC.
435  absl::optional<uint32_t> ssrc;
436 
437  // The relative bitrate priority of this encoding. Currently this is
438  // implemented for the entire rtp sender by using the value of the first
439  // encoding parameter.
440  // See: https://w3c.github.io/webrtc-priority/#enumdef-rtcprioritytype
441  // "very-low" = 0.5
442  // "low" = 1.0
443  // "medium" = 2.0
444  // "high" = 4.0
445  // TODO(webrtc.bugs.org/8630): Implement this per encoding parameter.
446  // Currently there is logic for how bitrate is distributed per simulcast layer
447  // in the VideoBitrateAllocator. This must be updated to incorporate relative
448  // bitrate priority.
449  double bitrate_priority = kDefaultBitratePriority;
450 
451  // The relative DiffServ Code Point priority for this encoding, allowing
452  // packets to be marked relatively higher or lower without affecting
453  // bandwidth allocations. See https://w3c.github.io/webrtc-dscp-exp/ .
454  // TODO(http://crbug.com/webrtc/8630): Implement this per encoding parameter.
455  // TODO(http://crbug.com/webrtc/11379): TCP connections should use a single
456  // DSCP value even if shared by multiple senders; this is not implemented.
457  Priority network_priority = Priority::kLow;
458 
459  // If set, this represents the Transport Independent Application Specific
460  // maximum bandwidth defined in RFC3890. If unset, there is no maximum
461  // bitrate. Currently this is implemented for the entire rtp sender by using
462  // the value of the first encoding parameter.
463  //
464  // Just called "maxBitrate" in ORTC spec.
465  //
466  // TODO(deadbeef): With ORTC RtpSenders, this currently sets the total
467  // bandwidth for the entire bandwidth estimator (audio and video). This is
468  // just always how "b=AS" was handled, but it's not correct and should be
469  // fixed.
470  absl::optional<int> max_bitrate_bps;
471 
472  // Specifies the minimum bitrate in bps for video.
473  absl::optional<int> min_bitrate_bps;
474 
475  // Specifies the maximum framerate in fps for video.
476  absl::optional<double> max_framerate;
477 
478  // Specifies the number of temporal layers for video (if the feature is
479  // supported by the codec implementation).
480  // Screencast support is experimental.
481  absl::optional<int> num_temporal_layers;
482 
483  // For video, scale the resolution down by this factor.
484  absl::optional<double> scale_resolution_down_by;
485 
486  // https://w3c.github.io/webrtc-svc/#rtcrtpencodingparameters
487  absl::optional<std::string> scalability_mode;
488 
489  // Requested encode resolution.
490  //
491  // This field provides an alternative to `scale_resolution_down_by`
492  // that is not dependent on the video source.
493  //
494  // When setting requested_resolution it is not necessary to adapt the
495  // video source using OnOutputFormatRequest, since the VideoStreamEncoder
496  // will apply downscaling if necessary. requested_resolution will also be
497  // propagated to the video source, this allows downscaling earlier in the
498  // pipeline which can be beneficial if the source is consumed by multiple
499  // encoders, but is not strictly necessary.
500  //
501  // The `requested_resolution` is subject to resource adaptation.
502  //
503  // It is an error to set both `requested_resolution` and
504  // `scale_resolution_down_by`.
505  absl::optional<Resolution> requested_resolution;
506 
507  // For an RtpSender, set to true to cause this encoding to be encoded and
508  // sent, and false for it not to be encoded and sent. This allows control
509  // across multiple encodings of a sender for turning simulcast layers on and
510  // off.
511  // TODO(webrtc.bugs.org/8807): Updating this parameter will trigger an encoder
512  // reset, but this isn't necessarily required.
513  bool active = true;
514 
515  // Value to use for RID RTP header extension.
516  // Called "encodingId" in ORTC.
517  std::string rid;
518  bool request_key_frame = false;
519 
520  // Allow dynamic frame length changes for audio:
521  // https://w3c.github.io/webrtc-extensions/#dom-rtcrtpencodingparameters-adaptiveptime
522  bool adaptive_ptime = false;
523 
524  // Allow changing the used codec for this encoding.
525  absl::optional<RtpCodec> codec;
526 
527  bool operator==(const RtpEncodingParameters& o) const {
528  return ssrc == o.ssrc && bitrate_priority == o.bitrate_priority &&
529  network_priority == o.network_priority &&
530  max_bitrate_bps == o.max_bitrate_bps &&
531  min_bitrate_bps == o.min_bitrate_bps &&
532  max_framerate == o.max_framerate &&
533  num_temporal_layers == o.num_temporal_layers &&
534  scale_resolution_down_by == o.scale_resolution_down_by &&
535  active == o.active && rid == o.rid &&
536  adaptive_ptime == o.adaptive_ptime &&
537  requested_resolution == o.requested_resolution && codec == o.codec;
538  }
539  bool operator!=(const RtpEncodingParameters& o) const {
540  return !(*this == o);
541  }
542 };
