259 struct RtpPacketCounter {
260  RtpPacketCounter()
261  : header_bytes(0), payload_bytes(0), padding_bytes(0), packets(0) {}
262 
263  explicit RtpPacketCounter(const RtpPacket& packet);
264  explicit RtpPacketCounter(const RtpPacketToSend& packet_to_send);
265 
266  void Add(const RtpPacketCounter& other) {
267  header_bytes += other.header_bytes;
268  payload_bytes += other.payload_bytes;
269  padding_bytes += other.padding_bytes;
270  packets += other.packets;
271  total_packet_delay += other.total_packet_delay;
272  }
273 
274  bool operator==(const RtpPacketCounter& other) const {
275  return header_bytes == other.header_bytes &&
276  payload_bytes == other.payload_bytes &&
277  padding_bytes == other.padding_bytes && packets == other.packets &&
278  total_packet_delay == other.total_packet_delay;
279  }
280 
281  // Not inlined, since use of RtpPacket would result in circular includes.
282  void AddPacket(const RtpPacket& packet);
283  void AddPacket(const RtpPacketToSend& packet_to_send);
284 
285  size_t TotalBytes() const {
286  return header_bytes + payload_bytes + padding_bytes;
287  }
288 
289  size_t header_bytes; // Number of bytes used by RTP headers.
290  size_t payload_bytes; // Payload bytes, excluding RTP headers and padding.
291  size_t padding_bytes; // Number of padding bytes.
292  size_t packets; // Number of packets.
293  // The total delay of all `packets`. For RtpPacketToSend packets, this is
294  // `time_in_send_queue()`. For receive packets, this is zero.
295  webrtc::TimeDelta total_packet_delay = webrtc::TimeDelta::Zero();
296 };

298 // Data usage statistics for a (rtp) stream.
299 struct StreamDataCounters {
300  StreamDataCounters();
301 
302  void Add(const StreamDataCounters& other) {
303  transmitted.Add(other.transmitted);
304  retransmitted.Add(other.retransmitted);
305  fec.Add(other.fec);
306  if (other.first_packet_time < first_packet_time) {
307  // Use oldest time (excluding unsed value represented as plus infinity.
308  first_packet_time = other.first_packet_time;
309  }
310  }
311 
312  void MaybeSetFirstPacketTime(Timestamp now) {
313  if (first_packet_time == Timestamp::PlusInfinity()) {
314  first_packet_time = now;
315  }
316  }
317 
318  // Return time since first packet is send/received, or zero if such event
319  // haven't happen.
320  TimeDelta TimeSinceFirstPacket(Timestamp now) const {
321  return first_packet_time == Timestamp::PlusInfinity()
322  ? TimeDelta::Zero()
323  : now - first_packet_time;
324  }
325 
326  // Returns the number of bytes corresponding to the actual media payload (i.e.
327  // RTP headers, padding, retransmissions and fec packets are excluded).
328  // Note this function does not have meaning for an RTX stream.
329  size_t MediaPayloadBytes() const {
330  return transmitted.payload_bytes - retransmitted.payload_bytes -
331  fec.payload_bytes;
332  }
333 
334  // Time when first packet is sent/received.
335  Timestamp first_packet_time = Timestamp::PlusInfinity();
336 
337  RtpPacketCounter transmitted; // Number of transmitted packets/bytes.
338  RtpPacketCounter retransmitted; // Number of retransmitted packets/bytes.
339  RtpPacketCounter fec; // Number of redundancy packets/bytes.
340 };
