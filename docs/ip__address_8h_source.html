51 // Version-agnostic IP address class, wraps a union of in_addr and in6_addr.
52 class RTC_EXPORT IPAddress {
53  public:
54  IPAddress() : family_(AF_UNSPEC) { ::memset(&u_, 0, sizeof(u_)); }
55 
56  explicit IPAddress(const in_addr& ip4) : family_(AF_INET) {
57  memset(&u_, 0, sizeof(u_));
58  u_.ip4 = ip4;
59  }
60 
61  explicit IPAddress(const in6_addr& ip6) : family_(AF_INET6) { u_.ip6 = ip6; }
62 
63  explicit IPAddress(uint32_t ip_in_host_byte_order) : family_(AF_INET) {
64  memset(&u_, 0, sizeof(u_));
65  u_.ip4.s_addr = HostToNetwork32(ip_in_host_byte_order);
66  }
67 
68  IPAddress(const IPAddress& other) : family_(other.family_) {
69  ::memcpy(&u_, &other.u_, sizeof(u_));
70  }
71 
72  virtual ~IPAddress() {}

74  const IPAddress& operator=(const IPAddress& other) {
75  family_ = other.family_;
76  ::memcpy(&u_, &other.u_, sizeof(u_));
77  return *this;
78  }
