28 // This structure holds the info needed to update the packet send time header

42 // This structure holds meta information for the packet which is about to send
43 // over network.
44 struct RTC_EXPORT PacketOptions {
45  PacketOptions();
46  explicit PacketOptions(DiffServCodePoint dscp);
47  PacketOptions(const PacketOptions& other);
48  ~PacketOptions();
49 
50  DiffServCodePoint dscp = DSCP_NO_CHANGE;
51  // When used with RTP packets (for example, webrtc::PacketOptions), the value
52  // should be 16 bits. A value of -1 represents "not set".
53  int64_t packet_id = -1;
54  PacketTimeUpdateParams packet_time_params;
55  // PacketInfo is passed to SentPacket when signaling this packet is sent.
56  PacketInfo info_signaled_after_sent;
57  // True if this is a batchable packet. Batchable packets are collected at low
58  // levels and sent first when their AsyncPacketSocket receives a
59  // OnSendBatchComplete call.
60  bool batchable = false;
61  // True if this is the last packet of a batch.
62  bool last_packet_in_batch = false;
63 };

65 // Provides the ability to receive packets asynchronously. Sends are not
66 // buffered since it is acceptable to drop packets under high load.
67 class RTC_EXPORT AsyncPacketSocket : public sigslot::has_slots<> {
68  public:
69  enum State {
70  STATE_CLOSED,
71  STATE_BINDING,
72  STATE_BOUND,
73  STATE_CONNECTING,
74  STATE_CONNECTED
75  };
76 
77  AsyncPacketSocket() = default;
78  ~AsyncPacketSocket() override;
79 
80  AsyncPacketSocket(const AsyncPacketSocket&) = delete;
81  AsyncPacketSocket& operator=(const AsyncPacketSocket&) = delete;
82 
83  // Returns current local address. Address may be set to null if the
84  // socket is not bound yet (GetState() returns STATE_BINDING).
85  virtual SocketAddress GetLocalAddress() const = 0;
86 
87  // Returns remote address. Returns zeroes if this is not a client TCP socket.
88  virtual SocketAddress GetRemoteAddress() const = 0;
89 
90  // Send a packet.
91  virtual int Send(const void* pv, size_t cb, const PacketOptions& options) = 0;
92  virtual int SendTo(const void* pv,
93  size_t cb,
94  const SocketAddress& addr,
95  const PacketOptions& options) = 0;
96 
97  // Close the socket.
98  virtual int Close() = 0;
99 
100  // Returns current state of the socket.
101  virtual State GetState() const = 0;
102 
103  // Get/set options.
104  virtual int GetOption(Socket::Option opt, int* value) = 0;
105  virtual int SetOption(Socket::Option opt, int value) = 0;
106 
107  // Get/Set current error.
108  // TODO: Remove SetError().
109  virtual int GetError() const = 0;
110  virtual void SetError(int error) = 0;
111 
112  // Register a callback to be called when the socket is closed.
113  void SubscribeCloseEvent(
114  const void* removal_tag,
115  std::function<void(AsyncPacketSocket*, int)> callback);
116  void UnsubscribeCloseEvent(const void* removal_tag);
117 
118  // Emitted each time a packet is read. Used only for UDP and
119  // connected TCP sockets.
120  sigslot::signal5<AsyncPacketSocket*,
121  const char*,
122  size_t,
123  const SocketAddress&,
124  // TODO(bugs.webrtc.org/9584): Change to passing the int64_t
125  // timestamp by value.
126  const int64_t&>
127  SignalReadPacket;
128 
129  // Emitted each time a packet is sent.
130  sigslot::signal2<AsyncPacketSocket*, const SentPacket&> SignalSentPacket;
131 
132  // Emitted when the socket is currently able to send.
133  sigslot::signal1<AsyncPacketSocket*> SignalReadyToSend;
134 
135  // Emitted after address for the socket is allocated, i.e. binding
136  // is finished. State of the socket is changed from BINDING to BOUND
137  // (for UDP sockets).
138  sigslot::signal2<AsyncPacketSocket*, const SocketAddress&> SignalAddressReady;
139 
140  // Emitted for client TCP sockets when state is changed from
141  // CONNECTING to CONNECTED.
142  sigslot::signal1<AsyncPacketSocket*> SignalConnect;
143 
144  void NotifyClosedForTest(int err) { NotifyClosed(err); }
145 
146  protected:
147  // TODO(bugs.webrtc.org/11943): Remove after updating downstream code.
148  void SignalClose(AsyncPacketSocket* s, int err) {
149  RTC_DCHECK_EQ(s, this);
150  NotifyClosed(err);
151  }
152 
153  void NotifyClosed(int err) {
154  RTC_DCHECK_RUN_ON(&network_checker_);
155  on_close_.Send(this, err);
156  }
157 
158  RTC_NO_UNIQUE_ADDRESS webrtc::SequenceChecker network_checker_{
159  webrtc::SequenceChecker::kDetached};
160 
161  private:
162  webrtc::CallbackList<AsyncPacketSocket*, int> on_close_
163  RTC_GUARDED_BY(&network_checker_);
164 };
