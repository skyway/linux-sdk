25 namespace webrtc {
26 // The VideoEncoderFactoryTemplate supports encoders implementations given as
27 // template arguments.
28 //
29 // To include an encoder in the factory it requires three static members
30 // functions to be defined:
31 //
32 // // Returns the supported SdpVideoFormats this encoder can produce.
33 // static std::vector<SdpVideoFormat> SupportedFormats();
34 //
35 // // Creates an encoder instance for the given format.
36 // static std::unique_ptr<VideoEncoder>
37 // CreateEncoder(const SdpVideoFormat& format);
38 //
39 // // Returns true if the encoder supports the given scalability mode.
40 // static bool
41 // IsScalabilityModeSupported(ScalabilityMode scalability_mode);
42 //
43 // Note that the order of the template arguments matter as the factory will
44 // query/return the first encoder implementation supporting the given
45 // SdpVideoFormat.
46 template <typename... Ts>
47 class VideoEncoderFactoryTemplate : public VideoEncoderFactory {
48  public:
49  std::vector<SdpVideoFormat> GetSupportedFormats() const override {
50  return GetSupportedFormatsInternal<Ts...>();
51  }
52 
53  std::unique_ptr<VideoEncoder> CreateVideoEncoder(
54  const SdpVideoFormat& format) override {
55  // We fuzzy match the specified format for both valid and not so valid
56  // reasons. The valid reason is that there are many standardized codec
57  // specific fmtp parameters that have not been implemented, and in those
58  // cases we should not fail to instantiate an encoder just because we don't
59  // recognize the parameter. The not so valid reason is that we have started
60  // adding parameters completely unrelated to the SDP to the SdpVideoFormat.
61  // TODO(bugs.webrtc.org/13868): Remove FuzzyMatchSdpVideoFormat
62  absl::optional<SdpVideoFormat> matched =
63  FuzzyMatchSdpVideoFormat(GetSupportedFormats(), format);
64  return CreateVideoEncoderInternal<Ts...>(matched.value_or(format));
65  }
66 
67  CodecSupport QueryCodecSupport(
68  const SdpVideoFormat& format,
69  absl::optional<std::string> scalability_mode) const override {
70  return QueryCodecSupportInternal<Ts...>(format, scalability_mode);
71  }
72 
73  private:
74  bool IsFormatInList(
75  const SdpVideoFormat& format,
76  rtc::ArrayView<const SdpVideoFormat> supported_formats) const {
77  return absl::c_any_of(
78  supported_formats, [&](const SdpVideoFormat& supported_format) {
79  return supported_format.name == format.name &&
80  supported_format.parameters == format.parameters;
81  });
82  }
83 
84  template <typename V>
85  bool IsScalabilityModeSupported(
86  const absl::optional<std::string>& scalability_mode_string) const {
87  if (!scalability_mode_string.has_value()) {
88  return true;
89  }
90  absl::optional<ScalabilityMode> scalability_mode =
91  ScalabilityModeFromString(*scalability_mode_string);
92  return scalability_mode.has_value() &&
93  V::IsScalabilityModeSupported(*scalability_mode);
94  }
95 
96  template <typename V, typename... Vs>
97  std::vector<SdpVideoFormat> GetSupportedFormatsInternal() const {
98  auto supported_formats = V::SupportedFormats();
99 
100  if constexpr (sizeof...(Vs) > 0) {
101  // Supported formats may overlap between implementations, so duplicates
102  // should be filtered out.
103  for (const auto& other_format : GetSupportedFormatsInternal<Vs...>()) {
104  if (!IsFormatInList(other_format, supported_formats)) {
105  supported_formats.push_back(other_format);
106  }
107  }
108  }
109 
110  return supported_formats;
111  }
112 
113  template <typename V, typename... Vs>
114  std::unique_ptr<VideoEncoder> CreateVideoEncoderInternal(
115  const SdpVideoFormat& format) {
116  if (IsFormatInList(format, V::SupportedFormats())) {
117  return V::CreateEncoder(format);
118  }
119 
120  if constexpr (sizeof...(Vs) > 0) {
121  return CreateVideoEncoderInternal<Vs...>(format);
122  }
123 
124  return nullptr;
125  }
126 
127  template <typename V, typename... Vs>
128  CodecSupport QueryCodecSupportInternal(
129  const SdpVideoFormat& format,
130  const absl::optional<std::string>& scalability_mode) const {
131  if (IsFormatInList(format, V::SupportedFormats())) {
132  return {.is_supported = IsScalabilityModeSupported<V>(scalability_mode)};
133  }
134 
135  if constexpr (sizeof...(Vs) > 0) {
136  return QueryCodecSupportInternal<Vs...>(format, scalability_mode);
137  }
138 
139  return {.is_supported = false};
140  }
141 };
