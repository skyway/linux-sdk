75 // TODO(terelius): Compared to a generic 'Status' class, this
76 // class allows us additional information about the context
77 // in which the error occurred. This is currently limited to
78 // the source location (file and line), but we plan on adding
79 // information about the event and field name being parsed.
80 // If/when we start using absl::Status in WebRTC, consider
81 // whether payloads would be an appropriate alternative.
82 class RtcEventLogParseStatus {
83  template <typename T>
84  friend class RtcEventLogParseStatusOr;
85 
86  public:
87  static RtcEventLogParseStatus Success() { return RtcEventLogParseStatus(); }
88  static RtcEventLogParseStatus Error(absl::string_view error,
89  absl::string_view file,
90  int line) {
91  return RtcEventLogParseStatus(error, file, line);
92  }
93 
94  bool ok() const { return error_.empty(); }
95  ABSL_DEPRECATED("Use ok() instead") explicit operator bool() const {
96  return ok();
97  }
98 
99  std::string message() const { return error_; }
100 
101  private:
102  RtcEventLogParseStatus() : error_() {}
103  RtcEventLogParseStatus(absl::string_view error,
104  absl::string_view file,
105  int line)
106  : error_(std::string(error) + " (" + std::string(file) + ": " +
107  std::to_string(line) + ")") {}
108 
109  std::string error_;
110 };
