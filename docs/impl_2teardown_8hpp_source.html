63  void
64  operator()(
65  error_code ec = {},
66  std::size_t bytes_transferred = 0,
67  bool cont = true)
68  {
69  BOOST_ASIO_CORO_REENTER(*this)
70  {
71  nb_ = s_.non_blocking();
72  s_.non_blocking(true, ec);
73  if(ec)
74  goto upcall;
75  if(role_ == role_type::server)
76  s_.shutdown(net::socket_base::shutdown_send, ec);
77  if(ec)
78  goto upcall;
79  for(;;)
80  {
81  {
82  char buf[2048];
83  s_.read_some(net::buffer(buf), ec);
84  }
85  if(ec == net::error::would_block)
86  {
87  BOOST_ASIO_CORO_YIELD
88  s_.async_wait(
89  net::socket_base::wait_read,
90  beast::detail::bind_continuation(std::move(*this)));
91  continue;
92  }
93  if(ec)
94  {
95  if(ec != net::error::eof)
96  goto upcall;
97  ec = {};
98  break;
99  }
100  if(bytes_transferred == 0)
101  {
102  // happens sometimes
103  // https://github.com/boostorg/beast/issues/1373
104  break;
105  }
106  }
107  if(role_ == role_type::client)
108  s_.shutdown(net::socket_base::shutdown_send, ec);
109  if(ec)
110  goto upcall;
111  s_.close(ec);
112  upcall:
113  if(! cont)
114  {
115  BOOST_ASIO_CORO_YIELD
116  net::post(bind_front_handler(
117  std::move(*this), ec));
118  }
119  {
120  error_code ignored;
121  s_.non_blocking(nb_, ignored);
122  }
123  this->complete_now(ec);
124  }
125  }
126 };

132 template<class Protocol, class Executor>
133 void
134 teardown(
135  role_type role,
136  net::basic_stream_socket<
137  Protocol, Executor>& socket,
138  error_code& ec)
139 {
140  if(role == role_type::server)
141  socket.shutdown(
142  net::socket_base::shutdown_send, ec);
143  if(ec)
144  return;
145  for(;;)
146  {
147  char buf[2048];
148  auto const bytes_transferred =
149  socket.read_some(net::buffer(buf), ec);
150  if(ec)
151  {
152  if(ec != net::error::eof)
153  return;
154  ec = {};
155  break;
156  }
157  if(bytes_transferred == 0)
158  {
159  // happens sometimes
160  // https://github.com/boostorg/beast/issues/1373
161  break;
162  }
163  }
164  if(role == role_type::client)
165  socket.shutdown(
166  net::socket_base::shutdown_send, ec);
167  if(ec)
168  return;
169  socket.close(ec);
170 }
