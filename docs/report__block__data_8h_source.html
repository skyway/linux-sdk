20 // Represents fields and derived information received in RTCP report block
21 // attached to RTCP sender report or RTCP receiver report, as described in
22 // https://www.rfc-editor.org/rfc/rfc3550#section-6.4.1
23 class ReportBlockData {
24  public:
25  ReportBlockData() = default;
26 
27  ReportBlockData(const ReportBlockData&) = default;
28  ReportBlockData& operator=(const ReportBlockData&) = default;
29 
30  // The SSRC identifier for the originator of this report block,
31  // i.e. remote receiver of the RTP stream.
32  uint32_t sender_ssrc() const { return sender_ssrc_; }
33 
34  // The SSRC identifier of the source to which the information in this
35  // reception report block pertains, i.e. local sender of the RTP stream.
36  uint32_t source_ssrc() const { return source_ssrc_; }
37 
38  // The fraction of RTP data packets from 'source_ssrc()' lost since the
39  // previous report block was sent.
40  // Fraction loss in range [0.0, 1.0].
41  float fraction_lost() const { return fraction_lost_raw() / 256.0; }
42 
43  // Fraction loss as was written in the raw packet: range is [0, 255] where 0
44  // represents no loss, and 255 represents 99.6% loss (255/256 * 100%).
45  uint8_t fraction_lost_raw() const { return fraction_lost_raw_; }
46 
47  // The total number of RTP data packets from 'source_ssrc()' that have been
48  // lost since the beginning of reception. This number is defined to be the
49  // number of packets expected less the number of packets actually received,
50  // where the number of packets received includes any which are late or
51  // duplicates. Thus, packets that arrive late are not counted as lost, and the
52  // loss may be negative if there are duplicates.
53  int cumulative_lost() const { return cumulative_lost_; }
54 
55  // The low 16 bits contain the highest sequence number received in an RTP data
56  // packet from 'source_ssrc()', and the most significant 16 bits extend that
57  // sequence number with the corresponding count of sequence number cycles.
58  uint32_t extended_highest_sequence_number() const {
59  return extended_highest_sequence_number_;
60  }
61 
62  // An estimate of the statistical variance of the RTP data packet interarrival
63  // time, measured in RTP timestamp units. The interarrival jitter J is defined
64  // to be the mean deviation (smoothed absolute value) of the difference D in
65  // packet spacing at the receiver compared to the sender for a pair of
66  // packets.
67  uint32_t jitter() const { return jitter_; }
68 
69  // Jitter converted to common time units.
70  TimeDelta jitter(int rtp_clock_rate_hz) const;
71 
72  // Time in utc epoch (Jan 1st, 1970) the report block was received.
73  Timestamp report_block_timestamp_utc() const {
74  return report_block_timestamp_utc_;
75  }
76 
77  // Round Trip Time measurments for given (sender_ssrc, source_ssrc) pair.
78  // Min, max, sum, number of measurements are since beginning of the call.
79  TimeDelta last_rtt() const { return last_rtt_; }
80  TimeDelta sum_rtts() const { return sum_rtt_; }
81  size_t num_rtts() const { return num_rtts_; }
82  bool has_rtt() const { return num_rtts_ != 0; }
83 
84  void set_sender_ssrc(uint32_t ssrc) { sender_ssrc_ = ssrc; }
85  void set_source_ssrc(uint32_t ssrc) { source_ssrc_ = ssrc; }
86  void set_fraction_lost_raw(uint8_t lost) { fraction_lost_raw_ = lost; }
87  void set_cumulative_lost(int lost) { cumulative_lost_ = lost; }
88  void set_extended_highest_sequence_number(uint32_t sn) {
89  extended_highest_sequence_number_ = sn;
90  }
91  void set_jitter(uint32_t jitter) { jitter_ = jitter; }
92  void set_report_block_timestamp_utc(Timestamp arrival_time) {
93  report_block_timestamp_utc_ = arrival_time;
94  }
95 
96  void SetReportBlock(uint32_t sender_ssrc,
97  const rtcp::ReportBlock& report_block,
98  Timestamp report_block_timestamp_utc);
99  void AddRoundTripTimeSample(TimeDelta rtt);
100 
101  private:
102  uint32_t sender_ssrc_ = 0;
103  uint32_t source_ssrc_ = 0;
104  uint8_t fraction_lost_raw_ = 0;
105  int32_t cumulative_lost_ = 0;
106  uint32_t extended_highest_sequence_number_ = 0;
107  uint32_t jitter_ = 0;
108  Timestamp report_block_timestamp_utc_ = Timestamp::Zero();
109  TimeDelta last_rtt_ = TimeDelta::Zero();
110  TimeDelta sum_rtt_ = TimeDelta::Zero();
111  size_t num_rtts_ = 0;
112 };
