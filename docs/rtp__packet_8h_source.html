233 template <typename Extension, typename FirstValue, typename... Values>
234 bool RtpPacket::GetExtension(FirstValue first, Values... values) const {
235  auto raw = FindExtension(Extension::kId);
236  if (raw.empty())
237  return false;
238  return Extension::Parse(raw, first, values...);
239 }

241 template <typename Extension>
242 absl::optional<typename Extension::value_type> RtpPacket::GetExtension() const {
243  absl::optional<typename Extension::value_type> result;
244  auto raw = FindExtension(Extension::kId);
245  if (raw.empty() || !Extension::Parse(raw, &result.emplace()))
246  result = absl::nullopt;
247  return result;
248 }

255 template <typename Extension, typename... Values>
256 bool RtpPacket::SetExtension(const Values&... values) {
257  const size_t value_size = Extension::ValueSize(values...);
258  auto buffer = AllocateExtension(Extension::kId, value_size);
259  if (buffer.empty())
260  return false;
261  return Extension::Write(buffer, values...);
262 }

264 template <typename Extension>
265 bool RtpPacket::SetRawExtension(rtc::ArrayView<const uint8_t> data) {
266  rtc::ArrayView<uint8_t> buffer =
267  AllocateExtension(Extension::kId, data.size());
268  if (buffer.empty()) {
269  return false;
270  }
271  std::memcpy(buffer.data(), data.data(), data.size());
272  return true;
273 }

275 template <typename Extension>
276 bool RtpPacket::ReserveExtension() {
277  auto buffer = AllocateExtension(Extension::kId, Extension::kValueSizeBytes);
278  if (buffer.empty())
279  return false;
280  memset(buffer.data(), 0, Extension::kValueSizeBytes);
281  return true;
282 }
