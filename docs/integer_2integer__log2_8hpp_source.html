29  template <typename T>
30  int integer_log2_impl(T x, int n) {
31 
32  int result = 0;
33 
34  while (x != 1) {
35 
36  const T t = static_cast<T>(x >> n);
37  if (t) {
38  result += n;
39  x = t;
40  }
41  n /= 2;
42 
43  }
44 
45  return result;
46  }
