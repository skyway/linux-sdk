88 // Negotiate a permessage-deflate client offer
89 //
90 template<class Allocator>
91 void
92 pmd_negotiate(
93  http::basic_fields<Allocator>& fields,
94  pmd_offer& config,
95  pmd_offer const& offer,
96  permessage_deflate const& o)
97 {
98  if(! (offer.accept && o.server_enable))
99  {
100  config.accept = false;
101  return;
102  }
103  config.accept = true;
104 
105  auto s = detail::pmd_negotiate_impl(config, offer, o);
106  if(config.accept)
107  fields.set(http::field::sec_websocket_extensions, s);
108 }
