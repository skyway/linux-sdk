33 // Abstract base class for RTCStats-derived dictionaries, see
34 // https://w3c.github.io/webrtc-stats/.
35 //
36 // All derived classes must have the following static variable defined:
37 // static const char kType[];
38 // It is used as a unique class identifier and a string representation of the
39 // class type, see https://w3c.github.io/webrtc-stats/#rtcstatstype-str*.
40 // Use the `WEBRTC_RTCSTATS_IMPL` macro when implementing subclasses, see macro
41 // for details.
42 //
43 // Derived classes list their dictionary members, RTCStatsMember<T>, as public
44 // fields, allowing the following:
45 //
46 // RTCFooStats foo("fooId", Timestamp::Micros(GetCurrentTime()));
47 // foo.bar = 42;
48 // foo.baz = std::vector<std::string>();
49 // foo.baz->push_back("hello world");
50 // uint32_t x = *foo.bar;
51 //
52 // Pointers to all the members are available with `Members`, allowing iteration:
53 //
54 // for (const RTCStatsMemberInterface* member : foo.Members()) {
55 // printf("%s = %s\n", member->name(), member->ValueToString().c_str());
56 // }
57 class RTC_EXPORT RTCStats {
58  public:
59  RTCStats(const std::string& id, Timestamp timestamp)
60  : id_(id), timestamp_(timestamp) {}
61 
62  virtual ~RTCStats() {}
63 
64  virtual std::unique_ptr<RTCStats> copy() const = 0;
65 
66  const std::string& id() const { return id_; }
67  // Time relative to the UNIX epoch (Jan 1, 1970, UTC), in microseconds.
68  Timestamp timestamp() const { return timestamp_; }
69 
70  // Returns the static member variable `kType` of the implementing class.
71  virtual const char* type() const = 0;
72  // Returns a vector of pointers to all the `RTCStatsMemberInterface` members
73  // of this class. This allows for iteration of members. For a given class,
74  // `Members` always returns the same members in the same order.
75  std::vector<const RTCStatsMemberInterface*> Members() const;
76  // Checks if the two stats objects are of the same type and have the same
77  // member values. Timestamps are not compared. These operators are exposed for
78  // testing.
79  bool operator==(const RTCStats& other) const;
80  bool operator!=(const RTCStats& other) const;
81 
82  // Creates a JSON readable string representation of the stats
83  // object, listing all of its members (names and values).
84  std::string ToJson() const;
85 
86  // Downcasts the stats object to an `RTCStats` subclass `T`. DCHECKs that the
87  // object is of type `T`.
88  template <typename T>
89  const T& cast_to() const {
90  RTC_DCHECK_EQ(type(), T::kType);
91  return static_cast<const T&>(*this);
92  }
93 
94  protected:
95  // Gets a vector of all members of this `RTCStats` object, including members
96  // derived from parent classes. `additional_capacity` is how many more members
97  // shall be reserved in the vector (so that subclasses can allocate a vector
98  // with room for both parent and child members without it having to resize).
99  virtual std::vector<const RTCStatsMemberInterface*>
100  MembersOfThisObjectAndAncestors(size_t additional_capacity) const;
101 
102  std::string const id_;
103  Timestamp timestamp_;
104 };

106 // All `RTCStats` classes should use these macros.
107 // `WEBRTC_RTCSTATS_DECL` is placed in a public section of the class definition.
108 // `WEBRTC_RTCSTATS_IMPL` is placed outside the class definition (in a .cc).
109 //
110 // These macros declare (in _DECL) and define (in _IMPL) the static `kType` and
111 // overrides methods as required by subclasses of `RTCStats`: `copy`, `type` and
112 // `MembersOfThisObjectAndAncestors`. The |...| argument is a list of addresses
113 // to each member defined in the implementing class. The list must have at least
114 // one member.
115 //
116 // (Since class names need to be known to implement these methods this cannot be
117 // part of the base `RTCStats`. While these methods could be implemented using
118 // templates, that would only work for immediate subclasses. Subclasses of
119 // subclasses also have to override these methods, resulting in boilerplate
120 // code. Using a macro avoids this and works for any `RTCStats` class, including
121 // grandchildren.)
122 //
123 // Sample usage:
124 //
125 // rtcfoostats.h:
126 // class RTCFooStats : public RTCStats {
127 // public:
128 // WEBRTC_RTCSTATS_DECL();
129 //
130 // RTCFooStats(const std::string& id, Timestamp timestamp);
131 //
132 // RTCStatsMember<int32_t> foo;
133 // RTCStatsMember<int32_t> bar;
134 // };
135 //
136 // rtcfoostats.cc:
137 // WEBRTC_RTCSTATS_IMPL(RTCFooStats, RTCStats, "foo-stats"
138 // &foo,
139 // &bar);
140 //
141 // RTCFooStats::RTCFooStats(const std::string& id, Timestamp timestamp)
142 // : RTCStats(id, timestamp),
143 // foo("foo"),
144 // bar("bar") {
145 // }
146 //
147 #define WEBRTC_RTCSTATS_DECL() \
148  protected: \
149  std::vector<const webrtc::RTCStatsMemberInterface*> \
150  MembersOfThisObjectAndAncestors(size_t local_var_additional_capacity) \
151  const override; \
152  \
153  public: \
154  static const char kType[]; \
155  \
156  std::unique_ptr<webrtc::RTCStats> copy() const override; \
157  const char* type() const override
158 
159 #define WEBRTC_RTCSTATS_IMPL(this_class, parent_class, type_str, ...) \
160  const char this_class::kType[] = type_str; \
161  \
162  std::unique_ptr<webrtc::RTCStats> this_class::copy() const { \
163  return std::make_unique<this_class>(*this); \
164  } \
165  \
166  const char* this_class::type() const { \
167  return this_class::kType; \
168  } \
169  \
170  std::vector<const webrtc::RTCStatsMemberInterface*> \
171  this_class::MembersOfThisObjectAndAncestors( \
172  size_t local_var_additional_capacity) const { \
173  const webrtc::RTCStatsMemberInterface* local_var_members[] = { \
174  __VA_ARGS__}; \
175  size_t local_var_members_count = \
176  sizeof(local_var_members) / sizeof(local_var_members[0]); \
177  std::vector<const webrtc::RTCStatsMemberInterface*> \
178  local_var_members_vec = parent_class::MembersOfThisObjectAndAncestors( \
179  local_var_members_count + local_var_additional_capacity); \
180  RTC_DCHECK_GE( \
181  local_var_members_vec.capacity() - local_var_members_vec.size(), \
182  local_var_members_count + local_var_additional_capacity); \
183  local_var_members_vec.insert(local_var_members_vec.end(), \
184  &local_var_members[0], \
185  &local_var_members[local_var_members_count]); \
186  return local_var_members_vec; \
187  }

209 // Interface for `RTCStats` members, which have a name and a value of a type
210 // defined in a subclass. Only the types listed in `Type` are supported, these
211 // are implemented by `RTCStatsMember<T>`. The value of a member may be
212 // undefined, the value can only be read if `is_defined`.
213 class RTCStatsMemberInterface {
214  public:
215  // Member value types.
216  enum Type {
217  kBool, // bool
218  kInt32, // int32_t
219  kUint32, // uint32_t
220  kInt64, // int64_t
221  kUint64, // uint64_t
222  kDouble, // double
223  kString, // std::string
224 
225  kSequenceBool, // std::vector<bool>
226  kSequenceInt32, // std::vector<int32_t>
227  kSequenceUint32, // std::vector<uint32_t>
228  kSequenceInt64, // std::vector<int64_t>
229  kSequenceUint64, // std::vector<uint64_t>
230  kSequenceDouble, // std::vector<double>
231  kSequenceString, // std::vector<std::string>
232 
233  kMapStringUint64, // std::map<std::string, uint64_t>
234  kMapStringDouble, // std::map<std::string, double>
235  };
236 
237  virtual ~RTCStatsMemberInterface() {}
238 
239  const char* name() const { return name_; }
240  virtual Type type() const = 0;
241  virtual bool is_sequence() const = 0;
242  virtual bool is_string() const = 0;
243  virtual bool is_defined() const = 0;
244  // Type and value comparator. The names are not compared. These operators are
245  // exposed for testing.
246  bool operator==(const RTCStatsMemberInterface& other) const {
247  return IsEqual(other);
248  }
249  bool operator!=(const RTCStatsMemberInterface& other) const {
250  return !(*this == other);
251  }
252  virtual std::string ValueToString() const = 0;
253  // This is the same as ValueToString except for kInt64 and kUint64 types,
254  // where the value is represented as a double instead of as an integer.
255  // Since JSON stores numbers as floating point numbers, very large integers
256  // cannot be accurately represented, so we prefer to display them as doubles
257  // instead.
258  virtual std::string ValueToJson() const = 0;
259 
260  template <typename T>
261  const T& cast_to() const {
262  RTC_DCHECK_EQ(type(), T::StaticType());
263  return static_cast<const T&>(*this);
264  }
265 
266  protected:
267  explicit RTCStatsMemberInterface(const char* name) : name_(name) {}
268 
269  virtual bool IsEqual(const RTCStatsMemberInterface& other) const = 0;
270 
271  const char* const name_;
272 };

274 // Template implementation of `RTCStatsMemberInterface`.
275 // The supported types are the ones described by
276 // `RTCStatsMemberInterface::Type`.
277 template <typename T>
278 class RTCStatsMember : public RTCStatsMemberInterface {
279  public:
280  explicit RTCStatsMember(const char* name)
281  : RTCStatsMemberInterface(name), value_() {}
282  RTCStatsMember(const char* name, const T& value)
283  : RTCStatsMemberInterface(name), value_(value) {}
284  RTCStatsMember(const char* name, T&& value)
285  : RTCStatsMemberInterface(name), value_(std::move(value)) {}
286  explicit RTCStatsMember(const RTCStatsMember<T>& other)
287  : RTCStatsMemberInterface(other.name_), value_(other.value_) {}
288  explicit RTCStatsMember(RTCStatsMember<T>&& other)
289  : RTCStatsMemberInterface(other.name_), value_(std::move(other.value_)) {}
290 
291  static Type StaticType();
292  Type type() const override { return StaticType(); }
293  bool is_sequence() const override;
294  bool is_string() const override;
295  bool is_defined() const override { return value_.has_value(); }
296  std::string ValueToString() const override;
297  std::string ValueToJson() const override;
298 
299  template <typename U>
300  inline T ValueOrDefault(U default_value) const {
301  return value_.value_or(default_value);
302  }
303 
304  // Assignment operators.
305  T& operator=(const T& value) {
306  value_ = value;
307  return value_.value();
308  }
309  T& operator=(const T&& value) {
310  value_ = std::move(value);
311  return value_.value();
312  }
313 
314  // Getter methods that look the same as absl::optional<T>. Please prefer these
315  // in order to unblock replacing RTCStatsMember<T> with absl::optional<T> in
316  // the future (https://crbug.com/webrtc/15164).
317  bool has_value() const { return value_.has_value(); }
318  const T& value() const { return value_.value(); }
319  T& value() { return value_.value(); }
320  T& operator*() {
321  RTC_DCHECK(value_);
322  return *value_;
323  }
324  const T& operator*() const {
325  RTC_DCHECK(value_);
326  return *value_;
327  }
328  T* operator->() {
329  RTC_DCHECK(value_);
330  return &(*value_);
331  }
332  const T* operator->() const {
333  RTC_DCHECK(value_);
334  return &(*value_);
335  }
336 
337  protected:
338  bool IsEqual(const RTCStatsMemberInterface& other) const override {
339  if (type() != other.type())
340  return false;
341  const RTCStatsMember<T>& other_t =
342  static_cast<const RTCStatsMember<T>&>(other);
343  return value_ == other_t.value_;
344  }
345 
346  private:
347  absl::optional<T> value_;
348 };
