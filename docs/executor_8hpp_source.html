307  // Helper function to check and return the implementation pointer.
308  impl_base* get_impl() const
309  {
310  if (!impl_)
311  {
312  bad_executor ex;
313  boost::asio::detail::throw_exception(ex);
314  }
315  return impl_;
316  }
317 
318  // Helper function to clone another implementation.
319  impl_base* clone() const BOOST_ASIO_NOEXCEPT
320  {
321  return impl_ ? impl_->clone() : 0;
322  }
323 
324  // Helper function to destroy an implementation.
325  void destroy() BOOST_ASIO_NOEXCEPT
326  {
327  if (impl_)
328  impl_->destroy();
329  }
