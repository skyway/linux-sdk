87 template <class T>
88 static std::string ToStringIfSet(const char* key,
89  const absl::optional<T>& val) {
90  std::string str;
91  if (val) {
92  str = key;
93  str += ": ";
94  str += val ? rtc::ToString(*val) : "";
95  str += ", ";
96  }
97  return str;
98 }

100 template <class T>
101 static std::string VectorToString(const std::vector<T>& vals) {
102  rtc::StringBuilder ost; // no-presubmit-check TODO(webrtc:8982)
103  ost << "[";
104  for (size_t i = 0; i < vals.size(); ++i) {
105  if (i > 0) {
106  ost << ", ";
107  }
108  ost << vals[i].ToString();
109  }
110  ost << "]";
111  return ost.Release();
112 }

114 // Options that can be applied to a VideoMediaChannel or a VideoMediaEngine.
115 // Used to be flags, but that makes it hard to selectively apply options.
116 // We are moving all of the setting of options to structs like this,
117 // but some things currently still use flags.
118 struct VideoOptions {
119  VideoOptions();
120  ~VideoOptions();
121 
122  void SetAll(const VideoOptions& change) {
123  SetFrom(&video_noise_reduction, change.video_noise_reduction);
124  SetFrom(&screencast_min_bitrate_kbps, change.screencast_min_bitrate_kbps);
125  SetFrom(&is_screencast, change.is_screencast);
126  }
127 
128  bool operator==(const VideoOptions& o) const {
129  return video_noise_reduction == o.video_noise_reduction &&
130  screencast_min_bitrate_kbps == o.screencast_min_bitrate_kbps &&
131  is_screencast == o.is_screencast;
132  }
133  bool operator!=(const VideoOptions& o) const { return !(*this == o); }
134 
135  std::string ToString() const {
136  rtc::StringBuilder ost;
137  ost << "VideoOptions {";
138  ost << ToStringIfSet("noise reduction", video_noise_reduction);
139  ost << ToStringIfSet("screencast min bitrate kbps",
140  screencast_min_bitrate_kbps);
141  ost << ToStringIfSet("is_screencast ", is_screencast);
142  ost << "}";
143  return ost.Release();
144  }

146  // Enable denoising? This flag comes from the getUserMedia
147  // constraint 'googNoiseReduction', and WebRtcVideoEngine passes it
148  // on to the codec options. Disabled by default.
149  absl::optional<bool> video_noise_reduction;
150  // Force screencast to use a minimum bitrate. This flag comes from
151  // the PeerConnection constraint 'googScreencastMinBitrate'. It is
152  // copied to the encoder config by WebRtcVideoChannel.
153  // TODO(https://crbug.com/1315155): Remove the ability to set it in Chromium

155  absl::optional<int> screencast_min_bitrate_kbps;
156  // Set by screencast sources. Implies selection of encoding settings
157  // suitable for screencast. Most likely not the right way to do
158  // things, e.g., screencast of a text document and screencast of a
159  // youtube video have different needs.
160  absl::optional<bool> is_screencast;
161  webrtc::VideoTrackInterface::ContentHint content_hint;
162 
163  private:

325 // The stats information is structured as follows:
326 // Media are represented by either MediaSenderInfo or MediaReceiverInfo.
