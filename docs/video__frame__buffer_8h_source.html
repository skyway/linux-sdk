32 // Base class for frame buffers of different types of pixel format and storage.
33 // The tag in type() indicates how the data is represented, and each type is
34 // implemented as a subclass. To access the pixel data, call the appropriate
35 // GetXXX() function, where XXX represents the type. There is also a function
36 // ToI420() that returns a frame buffer in I420 format, converting from the
37 // underlying representation if necessary. I420 is the most widely accepted
38 // format and serves as a fallback for video sinks that can only handle I420,
39 // e.g. the internal WebRTC software encoders. A special enum value 'kNative' is
40 // provided for external clients to implement their own frame buffer
41 // representations, e.g. as textures. The external client can produce such
42 // native frame buffers from custom video sources, and then cast it back to the
43 // correct subclass in custom video sinks. The purpose of this is to improve
44 // performance by providing an optimized path without intermediate conversions.
45 // Frame metadata such as rotation and timestamp are stored in
46 // webrtc::VideoFrame, and not here.
47 class RTC_EXPORT VideoFrameBuffer : public rtc::RefCountInterface {
48  public:
49  // New frame buffer types will be added conservatively when there is an
50  // opportunity to optimize the path between some pair of video source and
51  // video sink.
52  // GENERATED_JAVA_ENUM_PACKAGE: org.webrtc
53  // GENERATED_JAVA_CLASS_NAME_OVERRIDE: VideoFrameBufferType
54  enum class Type {
55  kNative,
56  kI420,
57  kI420A,
58  kI422,
59  kI444,
60  kI010,
61  kI210,
62  kI410,
63  kNV12,
64  };
65 
66  // This function specifies in what pixel format the data is stored in.
67  virtual Type type() const = 0;
68 
69  // The resolution of the frame in pixels. For formats where some planes are
70  // subsampled, this is the highest-resolution plane.
71  virtual int width() const = 0;
72  virtual int height() const = 0;
73 
74  // Returns a memory-backed frame buffer in I420 format. If the pixel data is
75  // in another format, a conversion will take place. All implementations must
76  // provide a fallback to I420 for compatibility with e.g. the internal WebRTC
77  // software encoders.
78  // Conversion may fail, for example if reading the pixel data from a texture
79  // fails. If the conversion fails, nullptr is returned.
80  virtual rtc::scoped_refptr<I420BufferInterface> ToI420() = 0;
81 
82  // GetI420() methods should return I420 buffer if conversion is trivial, i.e
83  // no change for binary data is needed. Otherwise these methods should return
84  // nullptr. One example of buffer with that property is
85  // WebrtcVideoFrameAdapter in Chrome - it's I420 buffer backed by a shared
86  // memory buffer. Therefore it must have type kNative. Yet, ToI420()
87  // doesn't affect binary data at all. Another example is any I420A buffer.
88  // TODO(https://crbug.com/webrtc/12021): Make this method non-virtual and
89  // behave as the other GetXXX methods below.
90  virtual const I420BufferInterface* GetI420() const;
91 
92  // A format specific scale function. Default implementation works by
93  // converting to I420. But more efficient implementations may override it,
94  // especially for kNative.
95  // First, the image is cropped to `crop_width` and `crop_height` and then
96  // scaled to `scaled_width` and `scaled_height`.
97  virtual rtc::scoped_refptr<VideoFrameBuffer> CropAndScale(int offset_x,
98  int offset_y,
99  int crop_width,
100  int crop_height,
101  int scaled_width,
102  int scaled_height);
103 
104  // Alias for common use case.
105  rtc::scoped_refptr<VideoFrameBuffer> Scale(int scaled_width,
106  int scaled_height) {
107  return CropAndScale(0, 0, width(), height(), scaled_width, scaled_height);
108  }
109 
110  // These functions should only be called if type() is of the correct type.
111  // Calling with a different type will result in a crash.
112  const I420ABufferInterface* GetI420A() const;
113  const I422BufferInterface* GetI422() const;
114  const I444BufferInterface* GetI444() const;
115  const I010BufferInterface* GetI010() const;
116  const I210BufferInterface* GetI210() const;
117  const I410BufferInterface* GetI410() const;
118  const NV12BufferInterface* GetNV12() const;
119 
120  // From a kNative frame, returns a VideoFrameBuffer with a pixel format in
121  // the list of types that is in the main memory with a pixel perfect
122  // conversion for encoding with a software encoder. Returns nullptr if the
123  // frame type is not supported, mapping is not possible, or if the kNative
124  // frame has not implemented this method. Only callable if type() is kNative.
125  virtual rtc::scoped_refptr<VideoFrameBuffer> GetMappedFrameBuffer(
126  rtc::ArrayView<Type> types);
127 
128  protected:
129  ~VideoFrameBuffer() override {}
130 };
