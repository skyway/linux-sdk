75  T* TryAddStats(std::unique_ptr<T> stats) {
76  T* stats_ptr = stats.get();
77  if (!stats_

87  // Gets the stat object of type `T` by ID, where `T` is any class descending
88  // from `RTCStats`.
89  // Returns null if there is no stats object for the given ID or it is the
90  // wrong type.
91  template <typename T>
92  const T* GetAs(const std::string& id) const {
93  const RTCStats* stats = Get(id);
94  if (!stats || stats->type() != T::kType) {
95  return nullptr;
96  }
97  return &stats->cast_to<const T>();
98  }

110  // Gets the subset of stats that are of type `T`, where `T` is any class
111  // descending from `RTCStats`.
112  template <typename T>
113  std::vector<const T*> GetStatsOfType() const {
114  std::vector<const T*> stats_of_type;
115  for (const RTCStats& stats : *this) {
116  if (stats.type() == T::kType)
117  stats_of_type.push_back(&stats.cast_to<const T>());
